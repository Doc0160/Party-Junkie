Les opérateurs logiques
ET && true && true = true, true && false = false
OU || true || true = true, true || false = true, false || false = false
EGALE == true == true = true, true == false = false
N'est pas EGALE != true != true = false, true != false = true
SUPERIEUR OU EGALE >= 5 >= 2 = true, 5 >= 6 = false
INFERIEUR OU EGALE <= 5 <= 2 = false, 5 <= 6 = true
